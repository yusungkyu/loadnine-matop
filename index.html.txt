<script>
document.addEventListener('DOMContentLoaded', () => {
    const SUPABASE_URL = 'https://your-project-url.supabase.co';
    const SUPABASE_KEY = 'your-anon-key';
    const supabaseClient = supabase.createClient(SUPABASE_URL, SUPABASE_KEY);

    async function initApp() {
        // 데이터 로드
        const { data, error } = await supabaseClient
            .from('user_tracking')
            .select('*')
            .order('created_at', { ascending: false });

        if (data && data.length > 0) {
            // 팝업에 드레스코드 연동
            document.getElementById('modal-dresscode-info').innerText = `최신 입장 유저의 드레스코드: ${data[0].dress_code || '미지정'}`;
            renderList(data);
        } else {
            document.getElementById('modal-dresscode-info').innerText = "현재 기록된 유저 데이터가 없습니다.";
        }
        
        setupDragEngine();
    }

    function renderList(items) {
        const logContainer = document.getElementById('log-container');
        logContainer.innerHTML = ''; 
        items.forEach(item => logContainer.appendChild(createLogElement(item)));
        if (items[0]) document.getElementById('user-depth-display').innerText = `최근 깊이: ${items[0].depth}m (유저: ${items[0].user_id})`;
    }

    function createLogElement(item) {
        const logItem = document.createElement('div');
        logItem.className = 'witch-drag-item'; 
        logItem.draggable = true; 
        logItem.id = `pin-${item.id}`;
        
        logItem.addEventListener('dragstart', () => logItem.classList.add('wt-dragging'));
        logItem.addEventListener('dragend', () => logItem.classList.remove('wt-dragging'));
        
        logItem.innerHTML = `
            <div class="witch-item-content">
                <span class="witch-blink-pin">📌</span>
                <span>유저 ID: ${item.user_id} | 깊이: ${item.depth}m | 드레스코드: ${item.dress_code || '미지정'}</span>
            </div>
            <button class="witch-delete-btn" onclick="deleteItem('${item.id}')">삭제</button>
        `;
        return logItem;
    }

    window.deleteItem = async (id) => {
        await supabaseClient.from('user_tracking').delete().eq('id', id);
        document.getElementById(`pin-${id}`)?.remove();
    };

    function setupDragEngine() {
        const logContainer = document.getElementById('log-container');
        logContainer.addEventListener('dragover', (e) => {
            e.preventDefault();
            const dragging = document.querySelector('.wt-dragging');
            const siblings = [...logContainer.querySelectorAll('.witch-drag-item:not(.wt-dragging)')];
            const next = siblings.find(sib => e.clientY <= sib.getBoundingClientRect().top + sib.offsetHeight / 2);
            logContainer.insertBefore(dragging, next);
        });
    }

    initApp();
});
