<script>
document.addEventListener('DOMContentLoaded', () => {
    const SUPABASE_URL = 'https://fmsjpwhuwsczhcjnhycg.supabase.co';
    const SUPABASE_KEY = 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6ImZtc2pwd2h1d3Njemhjam5oeWNnIiwicm9sZSI6ImFub24iLCJpYXQiOjE3Nzk5OTI0NjgsImV4cCI6MjA5NTU2ODQ2OH0.vb_-H7xVvcs0N-S2bf6iK7ZH6ryLxSl6uNhxHol-d5c';
    const isMatop = window.location.hostname.includes('matop');
    const PINS_TBL = isMatop ? 'pins_matop' : 'pins';
    const PROFILE_TBL = isMatop ? 'matop_profile' : 'tower_profile';
    
    let supabase = null;
    let currentBoard = 'floor1';
    let pins = [];
    let pendingPoint = null;
    let selectedPin = null;

    // 초기화
    if (window.supabase) {
        supabase = window.supabase.createClient(SUPABASE_URL, SUPABASE_KEY);
        loadData();
    }

    async function loadData() {
        const { data: p } = await supabase.from(PINS_TBL).select('*');
        const { data: pr } = await supabase.from(PROFILE_TBL).select('dress_code').eq('id', 1).maybeSingle();
        pins = p || [];
        document.getElementById('dressCodeText').textContent = pr?.dress_code || '미설정';
        render();
    }

    function render() {
        const list = document.getElementById('pinList');
        const layer = document.getElementById('pinLayer');
        const img = document.getElementById('bgImage');
        const rect = img.getBoundingClientRect();
        
        layer.innerHTML = '';
        list.innerHTML = '';
        
        const filtered = pins.filter(p => p.board_id === currentBoard);
        filtered.forEach((p, i) => {
            // 핀 렌더링
            const btn = document.createElement('button');
            btn.className = 'pin';
            btn.style.left = (p.x_ratio * rect.width) + 'px';
            btn.style.top = (p.y_ratio * rect.height) + 'px';
            btn.onclick = () => { 
                selectedPin = p; 
                document.getElementById('viewName').textContent = p.name.split(' / ')[0].replace('이름 : ', '');
                document.getElementById('viewWeapon').textContent = p.name.split(' / ')[1]?.replace('무기 : ', '') || '-';
                document.getElementById('viewModalBackdrop').style.display = 'flex';
            };
            layer.appendChild(btn);
            
            // 리스트 렌더링
            list.innerHTML += `<div class="pin-item"><span>${i+1}. ${p.name}</span><button class="btn btn-red" onclick="deletePin(${p.id})">삭제</button></div>`;
        });
    }

    // 전역 함수로 노출
    window.deletePin = async (id) => {
        await supabase.from(PINS_TBL).delete().eq('id', id);
        document.getElementById('viewModalBackdrop').style.display = 'none';
        loadData();
    };

    // 모달 이벤트
    document.querySelectorAll('.modal-backdrop').forEach(m => m.onclick = (e) => { if(e.target === m) m.style.display = 'none'; });
    document.getElementById('profileEditBtn').onclick = () => { document.getElementById('dressCodeInput').value = document.getElementById('dressCodeText').textContent; document.getElementById('profileModalBackdrop').style.display = 'flex'; };
    document.getElementById('profileSaveBtn').onclick = async () => {
        const val = document.getElementById('dressCodeInput').value;
        await supabase.from(PROFILE_TBL).upsert({ id: 1, dress_code: val });
        loadData();
        document.getElementById('profileModalBackdrop').style.display = 'none';
    };
    document.getElementById('imageWrap').onclick = (e) => {
        if(e.target.id !== 'bgImage' && e.target.id !== 'pinLayer') return;
        const rect = e.target.getBoundingClientRect();
        pendingPoint = { x: (e.clientX - rect.left)/rect.width, y: (e.clientY - rect.top)/rect.height };
        document.getElementById('createModalBackdrop').style.display = 'flex';
    };
    document.getElementById('saveBtn').onclick = async () => {
        await supabase.from(PINS_TBL).insert({ 
            board_id: currentBoard, 
            x_ratio: pendingPoint.x, 
            y_ratio: pendingPoint.y, 
            name: `이름 : ${document.getElementById('pinName').value} / 무기 : ${document.getElementById('pinWeapon').value}` 
        });
        document.getElementById('createModalBackdrop').style.display = 'none';
        loadData();
    };
    document.querySelectorAll('.tab-btn').forEach(b => b.onclick = (e) => {
        document.querySelectorAll('.tab-btn').forEach(el => el.classList.remove('active'));
        e.target.classList.add('active');
        currentBoard = e.target.dataset.id;
        document.getElementById('bgImage').src = currentBoard === 'floor1' ? 'https://i.ibb.co/0RST75vn/1.png' : 'https://i.ibb.co/MD9V69Vm/2.png';
        render();
    });
    document.getElementById('refreshBtn').onclick = loadData;
});
