    <h2>🏰 검은마녀의탑 모니터링 대시보드</h2>
    
    <div class="panel">
        <h3>📍 실시간 유저 깊이(Depth) 현황</h3>
        <div id="user-depth-display"></div>
    </div>

    <div class="panel">
        <h3>📜 실시간 트랜잭션 로그 (드래그 정렬 가능)</h3>
        <div id="log-container"></div>

<script>
document.addEventListener('DOMContentLoaded', () => {
    
    // [필수 변수 설정] 본인의 실제 Supabase URL과 Anon KEY로 대체하세요
    const SUPABASE_URL = 'https://your-project-url.supabase.co';
    const SUPABASE_KEY = 'your-anon-key';

    let supabaseClient;

    // 1. Supabase 클라이언트 안전 초기화
    try {
        if (typeof supabase !== 'undefined') {
            supabaseClient = supabase.createClient(SUPABASE_URL, SUPABASE_KEY);
            console.log('✅ Supabase 클라이언트 초기화 성공');
            initApp();
        } else {
            throw new Error('Supabase 라이브러리 로드 실패');
        }
    } catch (error) {
        console.error('❌ 초기화 실패:', error.message);
        document.getElementById('user-depth-display').innerHTML = `<span class="error-msg">연결 실패: ${error.message}</span>`;
        return;
    }

    // 2. 핵심 서비스 구동
    function initApp() {
        fetchInitialData(); 
        setupDragAndDrop(); // 확실한 드래그 이벤트 엔진 가동
        
        supabaseClient
            .channel('witch-tower-changes')
            .on(
                'postgres_changes',
                { event: '*', schema: 'public', table: 'user_tracking' },
                (payload) => {
                    handleRealtimeUpdate(payload);
                }
            )
            .subscribe();
    }

    // 3. 서버 초기 데이터 조회
    async function fetchInitialData() {
        try {
            const { data, error } = await supabaseClient
                .from('user_tracking')
                .select('*')
                .order('created_at', { ascending: false })
                .limit(20);

            if (error) throw error;
            renderList(data);
        } catch (err) {
            console.error('❌ 데이터 로드 실패:', err.message);
        }
    }

    // 4. UI 리스트 출력
    function renderList(items) {
        const logContainer = document.getElementById('log-container');
        logContainer.innerHTML = ''; 

        if (!items || items.length === 0) return;
        
        items.forEach(item => {
            const logItem = createLogElement(item);
            logContainer.appendChild(logItem);
        });
        
        if (items[0]) updateDepthDisplay(items[0]);
    }

    // 5. 📌 개별 행 엘리먼트 생성 (드래그 속성 완벽 주입)
    function createLogElement(item) {
        const logItem = document.createElement('div');
        logItem.className = 'log-entry';
        logItem.id = `pin-${item.id}`; 
        logItem.draggable = true; // HTML5 드래그 기능 켜기

        // 드래그 액션 발생 시 클래스 부여/제거
        logItem.addEventListener('dragstart', () => logItem.classList.add('dragging'));
        logItem.addEventListener('dragend', () => logItem.classList.remove('dragging'));
        
        const contentWrap = document.createElement('div');
        contentWrap.className = 'log-content';

        const pinSpan = document.createElement('span');
        pinSpan.className = 'blinking-pin';
        pinSpan.innerText = '📌';

        const textSpan = document.createElement('span');
        textSpan.innerText = `유저 ID: ${item.user_id} | 깊이: ${item.depth}m`;
        
        contentWrap.appendChild(pinSpan);
        contentWrap.appendChild(textSpan);
        
        const deleteBtn = document.createElement('button');
        deleteBtn.className = 'delete-btn';
        deleteBtn.innerText = '삭제';
        
        // 🛑 [검증 완료] confirm()창 절대 없음. 클릭 시 바로 deletePinImmediately로 직행합니다.
        deleteBtn.onclick = (e) => {
            e.stopPropagation(); // 버튼 클릭 시 드래그 이벤트가 방해받지 않도록 차단
            deletePinImmediately(item.id);
        };

        logItem.appendChild(contentWrap);
        logItem.appendChild(deleteBtn);
        return logItem;
    }

    // 🔥 6. [오류 수정] 마우스 위치를 정밀 계산하여 부드럽게 정렬하는 드래그 엔진
    function setupDragAndDrop() {
        const logContainer = document.getElementById('log-container');
        
        logContainer.addEventListener('dragover', (e) => {
            e.preventDefault(); // 드롭 타깃으로 설정하기 위해 필수
            const draggingItem = document.querySelector('.dragging');
            if (!draggingItem) return;

            // 마우스 커서 위치와 가장 가까운 다른 행(Sibling) 계산
            const nextSibling = getDragAfterElement(logContainer, e.clientY);
            
            if (nextSibling == null) {
                logContainer.appendChild(draggingItem);
            } else {
                logContainer.insertBefore(draggingItem, nextSibling);
            }
        });
    }

    // 드래그 위치 정밀 계산용 수식 알고리즘
    function getDragAfterElement(container, y) {
        const draggableElements = [...container.querySelectorAll('.log-entry:not(.dragging)')];
        
        return draggableElements.reduce((closest, child) => {
            const box = child.getBoundingClientRect();
            const offset = y - box.top - box.height / 2; // 요소의 중심점 기준 거리 계산
            
            if (offset < 0 && offset > closest.offset) {
                return { offset: offset, element: child };
            } else {
                return closest;
            }
        }, { offset: Number.NEGATIVE_INFINITY }).element;
    }

    // 7. DB 즉시 삭제 처리 함수 (확인 팝업 전면 없음)
    async function deletePinImmediately(id) {
        try {
            const { error } = await supabaseClient
                .from('user_tracking')
                .delete()
                .eq('id', id);

            if (error) throw error;
            console.log(`📌 ID ${id} 항목이 즉각 삭제 처리되었습니다.`);
        } catch (err) {
            console.error('❌ DB 삭제 실패:', err.message);
        }
    }

    // 8. 실시간 스트리밍 핸들러
    function handleRealtimeUpdate(payload) {
        const logContainer = document.getElementById('log-container');
        const eventType = payload.eventType;

        if (eventType === 'INSERT') {
            const newElement = createLogElement(payload.new);
            logContainer.insertBefore(newElement, logContainer.firstChild);
            updateDepthDisplay(payload.new);
        } 
        else if (eventType === 'UPDATE') {
            const target = document.getElementById(`pin-${payload.new.id}`);
            if (target) {
                target.querySelector('.log-content span:nth-child(2)').innerText = `유저 ID: ${payload.new.user_id} | 깊이: ${payload.new.depth}m (수정됨)`;
            }
            updateDepthDisplay(payload.new);
        } 
        else if (eventType === 'DELETE') {
            const target = document.getElementById(`pin-${payload.old.id}`);
            if (target) target.remove();
        }
    }

    function updateDepthDisplay(item) {
        document.getElementById('user-depth-display').innerText = `최근 깊이: ${item.depth || 0}m (유저: ${item.user_id})`;
    }
});
