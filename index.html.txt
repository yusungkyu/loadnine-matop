<script src="https://cdn.jsdelivr.net/npm/@supabase/supabase-js@2/dist/umd/supabase.js"></script>
<script>
document.addEventListener('DOMContentLoaded', () => {
    const SUPABASE_URL = 'https://fmsjpwhuwsczhcjnhycg.supabase.co';
    const SUPABASE_KEY = 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6ImZtc2pwd2h1d3Njemhjam5oeWNnIiwicm9sZSI6ImFub24iLCJpYXQiOjE3Nzk5OTI0NjgsImV4cCI6MjA5NTU2ODQ2OH0.vb_-H7xVvcs0N-S2bf6iK7ZH6ryLxSl6uNhxHol-d5c';
    
    const isMatop = window.location.hostname.includes('matop');
    const PINS_TBL = isMatop ? 'pins_matop' : 'pins';
    const PROFILE_TBL = isMatop ? 'matop_profile' : 'tower_profile';
    
    const mainTitle = document.getElementById('mainTitle');
    if (mainTitle) mainTitle.textContent = isMatop ? '마력의 탑 위치 현황' : '검은 마녀의 탑 위치 현황';

    let supabase = null;
    let currentBoard = 'floor1';
    let pins = [];
    let pendingPoint = null;
    let selectedPin = null;

    if (window.supabase) {
        supabase = window.supabase.createClient(SUPABASE_URL, SUPABASE_KEY);
        loadData(true);
        
        // 실시간 연동 (데이터 변경 시 자동 새로고침)
        supabase.channel('realtime').on('postgres_changes', { event: '*', schema: 'public', table: PINS_TBL }, () => loadData(false)).subscribe();
        supabase.channel('realtime_profile').on('postgres_changes', { event: '*', schema: 'public', table: PROFILE_TBL }, () => loadData(false)).subscribe();
    }

    async function loadData(showIntro = false) {
        const { data: p } = await supabase.from(PINS_TBL).select('*').order('created_at', { ascending: true });
        const { data: pr } = await supabase.from(PROFILE_TBL).select('dress_code').eq('id', 1).maybeSingle();
        
        pins = p || [];
        const currentDressCode = pr?.dress_code || '미설정';
        
        document.getElementById('dressCodeText').textContent = currentDressCode;
        
        if (showIntro) {
            document.getElementById('introDressCodeText').textContent = currentDressCode;
            document.getElementById('introModalBackdrop').style.display = 'flex';
        }
        
        render();
    }

    function render() {
        const list = document.getElementById('pinList');
        const layer = document.getElementById('pinLayer');
        const img = document.getElementById('bgImage');
        const rect = img.getBoundingClientRect();
        
        document.getElementById('listTitle').textContent = currentBoard === 'floor1' ? '1층 핀 목록' : '2층 핀 목록';
        
        layer.innerHTML = '';
        list.innerHTML = '';
        
        const filtered = pins.filter(p => p.board_id === currentBoard);
        
        if (filtered.length === 0) {
            list.innerHTML = '<div style="color: #6b7280; font-size: 13px; padding: 6px 0;">등록된 핀이 없습니다.</div>';
        }

        filtered.forEach((p, i) => {
            const rawName = p.name || '';
            const displayStr = rawName.includes('이름 :') && rawName.includes(' / 무기 :') 
                ? rawName 
                : `이름 : ${rawName} / 무기 : 입력 안함`;
                
            const splitName = displayStr.split(' / ');
            const n = splitName[0].replace('이름 : ', '').trim();
            const w = (splitName[1] || '').replace('무기 : ', '').trim();

            // 1. 핀 그리기
            const btn = document.createElement('button');
            btn.className = 'pin';
            btn.style.left = (p.x_ratio * rect.width) + 'px';
            btn.style.top = (p.y_ratio * rect.height) + 'px';
            btn.onclick = (e) => { 
                e.stopPropagation();
                selectedPin = p; 
                document.getElementById('viewName').textContent = n;
                document.getElementById('viewWeapon').textContent = w;
                document.getElementById('viewModalBackdrop').style.display = 'flex';
            };
            layer.appendChild(btn);
            
            // 2. 핀 목록 & 바로 삭제 버튼 그리기
            list.innerHTML += `
                <div class="pin-item">
                    <span>${i+1}. ${displayStr}</span>
                    <button class="btn-delete-direct" onclick="deletePin(${p.id})">삭제</button>
                </div>
            `;
        });
    }

    // 전역 함수 (리스트에서 바로 삭제)
    window.deletePin = async (id) => {
        if (!confirm('이 핀을 삭제하시겠습니까?')) return;
        await supabase.from(PINS_TBL).delete().eq('id', id);
        loadData(false);
    };

    // 모달 배경 클릭 시 닫기
    document.querySelectorAll('.modal-backdrop').forEach(m => m.onclick = (e) => { 
        if (e.target === m) m.style.display = 'none'; 
    });

    // 버튼 이벤트 연결
    document.getElementById('introConfirmBtn').onclick = () => document.getElementById('introModalBackdrop').style.display = 'none';
    
    document.getElementById('cancelCreateBtn').onclick = () => document.getElementById('createModalBackdrop').style.display = 'none';
    document.getElementById('closeViewBtn').onclick = () => document.getElementById('viewModalBackdrop').style.display = 'none';
    document.getElementById('profileCancelBtn').onclick = () => document.getElementById('profileModalBackdrop').style.display = 'none';

    document.getElementById('profileEditBtn').onclick = () => { 
        document.getElementById('dressCodeInput').value = document.getElementById('dressCodeText').textContent; 
        document.getElementById('profileModalBackdrop').style.display = 'flex'; 
    };
    
    document.getElementById('profileSaveBtn').onclick = async () => {
        const val = document.getElementById('dressCodeInput').value;
        await supabase.from(PROFILE_TBL).upsert({ id: 1, dress_code: val });
        loadData(false);
        document.getElementById('profileModalBackdrop').style.display = 'none';
    };

    // 이미지 클릭 시 핀 등록
    document.getElementById('imageWrap').onclick = (e) => {
        if (e.target.id !== 'bgImage') return;
        const rect = e.target.getBoundingClientRect();
        pendingPoint = { 
            x: (e.clientX - rect.left) / rect.width, 
            y: (e.clientY - rect.top) / rect.height 
        };
        document.getElementById('pinName').value = '';
        document.getElementById('pinWeapon').selectedIndex = 0;
        document.getElementById('createModalBackdrop').style.display = 'flex';
    };

    // 핀 저장
    document.getElementById('saveBtn').onclick = async () => {
        if (!pendingPoint) return;
        const nameVal = document.getElementById('pinName').value.trim() || '무명';
        const weaponVal = document.getElementById('pinWeapon').value;
        
        await supabase.from(PINS_TBL).insert({ 
            board_id: currentBoard, 
            x_ratio: pendingPoint.x, 
            y_ratio: pendingPoint.y, 
            name: `이름 : ${nameVal} / 무기 : ${weaponVal}` 
        });
        document.getElementById('createModalBackdrop').style.display = 'none';
        loadData(false);
    };

    // 상세 보기 팝업에서 삭제
    document.getElementById('deleteBtn').onclick = async () => {
        if (!selectedPin || !confirm('이 핀을 삭제하시겠습니까?')) return;
        await supabase.from(PINS_TBL).delete().eq('id', selectedPin.id);
        document.getElementById('viewModalBackdrop').style.display = 'none';
        loadData(false);
    };

    // 탭 전환
    document.querySelectorAll('.tab-btn').forEach(b => b.onclick = (e) => {
        document.querySelectorAll('.tab-btn').forEach(el => el.classList.remove('active'));
        e.target.classList.add('active');
        currentBoard = e.target.dataset.id;
        document.getElementById('bgImage').src = currentBoard === 'floor1' ? 'https://i.ibb.co/0RST75vn/1.png' : 'https://i.ibb.co/MD9V69Vm/2.png';
        
        // 이미지 로드 후 렌더링을 위해 이벤트 리스너 추가
        document.getElementById('bgImage').onload = render;
        render(); // 캐시된 이미지를 위해 즉시 호출도 병행
    });

    // 새로고침 버튼
    document.getElementById('refreshBtn').onclick = () => loadData(false);
});
</script>
