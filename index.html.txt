document.addEventListener('DOMContentLoaded', async function () {
    const supabaseClient = supabase.createClient('URL', 'KEY'); // 키 입력 필요

    // 데이터 로드 및 팝업/화면 연동
    async function init() {
        const { data } = await supabaseClient.from('tower_profile').select('dress_code').eq('id', 1).maybeSingle();
        const code = data?.dress_code || '설정 필요';
        
        document.getElementById('dressCodeText').textContent = code;
        document.getElementById('introDressCodeText').textContent = code;
        document.getElementById('introModalBackdrop').style.display = 'flex';
    }

    document.getElementById('introConfirmBtn').addEventListener('click', () => {
        document.getElementById('introModalBackdrop').style.display = 'none';
    });

    init();
});
