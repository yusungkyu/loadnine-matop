<script>
document.addEventListener('DOMContentLoaded', () => {
    // Supabase 설정 데이터 연동
    const SUPABASE_URL = 'https://fmsjpwhuwsczhcjnhycg.supabase.co';
    const SUPABASE_KEY = 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6ImZtc2pwd2h1d3Njemhjam5oeWNnIiwicm9sZSI6ImFub24iLCJpYXQiOjE3Nzk5OTI0NjgsImV4cCI6MjA5NTU2ODQ2OH0.vb_-H7xVvcs0N-S2bf6iK7ZH6ryLxSl6uNhxHol-d5c';
    
    const isMatop = window.location.hostname.includes('matop');
    const PINS_TBL = isMatop ? 'pins_matop' : 'pins';
    const PROFILE_TBL = isMatop ? 'matop_profile' : 'tower_profile';
    
    const mainTitle = document.getElementById('mainTitle');
    if (mainTitle) mainTitle.textContent = isMatop ? '🏰 마력의 탑 모니터링 대시보드' : '🏰 검은마녀의탑 모니터링 대시보드';

    let currentBoard = 'floor1';
    let pins = [];
    let pendingPoint = null;
    let selectedPin = null;

    try {
        if (typeof supabase !== 'undefined') {
            supabaseClient = supabase.createClient(SUPABASE_URL, SUPABASE_KEY);
            console.log('✅ Supabase 클라이언트 초기화 성공');
            initApp();
        } else {
            throw new Error('Supabase 라이브러리가 로드되지 않았습니다.');
        }
    } catch (error) {
        console.error('❌ 초기화 실패:', error.message);
        document.getElementById('user-depth-display').innerHTML = `<span class="error-msg">연결 실패: ${error.message}</span>`;
        return;
    }

    function initApp() {
        // 지도 및 프로필 로드
        loadMapData(true);
        // 기존 유저 깊이 데이터 로드 (원본 로직)
        fetchInitialData();
        
        // 실시간 채널 통합 활성화
        supabaseClient.channel('map-changes').on('postgres_changes', { event: '*', schema: 'public', table: PINS_TBL }, () => loadMapData(false)).subscribe();
        supabaseClient.channel('profile-changes').on('postgres_changes', { event: '*', schema: 'public', table: PROFILE_TBL }, () => loadMapData(false)).subscribe();
        
        // 기존 트랙킹 실시간 채널 구독 (원본 로직)
        supabaseClient.channel('witch-tower-changes')
            .on('postgres_changes', { event: '*', schema: 'public', table: 'user_tracking' }, (payload) => {
                console.log('🔄 실시간 데이터 변동 감지:', payload);
                handleRealtimeUpdate(payload);
            })
            .subscribe((status) => {
                if (status === 'SUBSCRIBED') {
                    console.log('📡 실시간 트랙킹 로그 채널 구독 성공');
                }
            });
    }

    // 지도 및 핀 정보 데이터 호출
    async function loadMapData(showIntro = false) {
        const { data: p } = await supabaseClient.from(PINS_TBL).select('*').order('created_at', { ascending: true });
        const { data: pr } = await supabaseClient.from(PROFILE_TBL).select('dress_code').eq('id', 1).maybeSingle();
        
        pins = p || [];
        const currentDressCode = pr?.dress_code || '미설정';
        document.getElementById('dressCodeText').textContent = currentDressCode;
        
        if (showIntro) {
            document.getElementById('introDressCodeText').textContent = currentDressCode;
            document.getElementById('introModalBackdrop').style.display = 'flex';
        }
        renderMap();
    }

    // ⚡ 실시간 깜빡임 및 드래그 앤 드롭 핀 렌더링 시스템
    function renderMap() {
        const list = document.getElementById('pinList');
        const layer = document.getElementById('pinLayer');
        const img = document.getElementById('bgImage');
        const rect = img.getBoundingClientRect();
        
        document.getElementById('listTitle').textContent = currentBoard === 'floor1' ? '📍 1층 실시간 위치 목록' : '📍 2층 실시간 위치 목록';
        layer.innerHTML = '';
        list.innerHTML = '';
        
        const filtered = pins.filter(p => p.board_id === currentBoard);
        
        if (filtered.length === 0) {
            list.innerHTML = '<div style="color: #6b7280; font-size: 13px; padding: 6px 0;">등록된 실시간 위치가 없습니다.</div>';
        }

        filtered.forEach((p, i) => {
            const rawName = p.name || '';
            const displayStr = rawName.includes('이름 :') && rawName.includes(' / 무기 :') ? rawName : `이름 : ${rawName} / 무기 : 입력 안함`;
            const splitName = displayStr.split(' / ');
            const n = splitName[0].replace('이름 : ', '').trim();
            const w = (splitName[1] || '').replace('무기 : ', '').trim();

            const btn = document.createElement('button');
            btn.className = 'pin';
            btn.style.left = (p.x_ratio * rect.width) + 'px';
            btn.style.top = (p.y_ratio * rect.height) + 'px';
            
            // 드래그 기능 구현 (PC 마우스 및 모바일 터치 완벽 대응)
            const handleDragStart = (e) => {
                e.preventDefault();
                e.stopPropagation();
                let isDragging = false;
                const isTouch = e.type === 'touchstart';
                const moveEvent = isTouch ? 'touchmove' : 'mousemove';
                const upEvent = isTouch ? 'touchend' : 'mouseup';

                const onMove = (moveEv) => {
                    isDragging = true;
                    const clientX = isTouch ? moveEv.touches[0].clientX : moveEv.clientX;
                    const clientY = isTouch ? moveEv.touches[0].clientY : moveEv.clientY;
                    
                    let newX = clientX - img.getBoundingClientRect().left;
                    let newY = clientY - img.getBoundingClientRect().top;
                    
                    btn.style.left = Math.max(0, Math.min(newX, img.getBoundingClientRect().width)) + 'px';
                    btn.style.top = Math.max(0, Math.min(newY, img.getBoundingClientRect().height)) + 'px';
                };

                const onUp = async (upEv) => {
                    document.removeEventListener(moveEvent, onMove);
                    document.removeEventListener(upEvent, onUp);
                    
                    if (isDragging) {
                        const clientX = isTouch ? upEv.changedTouches[0].clientX : upEv.clientX;
                        const clientY = isTouch ? upEv.changedTouches[0].clientY : upEv.clientY;
                        
                        let finalXRatio = (clientX - img.getBoundingClientRect().left) / img.getBoundingClientRect().width;
                        let finalYRatio = (clientY - img.getBoundingClientRect().top) / img.getBoundingClientRect().height;
                        
                        await supabaseClient.from(PINS_TBL).update({
                            x_ratio: Math.max(0, Math.min(finalXRatio, 1)),
                            y_ratio: Math.max(0, Math.min(finalYRatio, 1))
                        }).eq('id', p.id);
                    } else {
                        selectedPin = p; 
                        document.getElementById('viewName').textContent = n;
                        document.getElementById('viewWeapon').textContent = w;
                        document.getElementById('viewModalBackdrop').style.display = 'flex';
                    }
                };

                document.addEventListener(moveEvent, onMove, { passive: false });
                document.addEventListener(upEvent, onUp);
            };

            btn.onmousedown = handleDragStart;
            btn.ontouchstart = handleDragStart;

            layer.appendChild(btn);
            list.innerHTML += `<div class="pin-item"><span>${i+1}. ${displayStr}</span><button class="btn-delete-direct" onclick="deletePin(${p.id})">삭제</button></div>`;
        });
    }

    // 핀 바로 삭제 기능
    window.deletePin = async (id) => {
        if (!confirm('이 위치 핀을 삭제하시겠습니까?')) return;
        await supabaseClient.from(PINS_TBL).delete().eq('id', id);
        loadMapData(false);
    };

    // 기존 원본 대시보드 로직 (유저 깊이 추적 및 트랜잭션 로그)
    async function fetchInitialData() {
        try {
            const { data, error } = await supabaseClient
                .from('user_tracking')
                .select('*')
                .order('created_at', { ascending: false })
                .limit(10);

            if (error) throw error;
            
            if (data && data.length > 0) {
                renderInitialList(data);
            } else {
                document.getElementById('user-depth-display').innerText = '현재 기록된 데이터가 없습니다.';
            }
        } catch (err) {
            console.error('❌ 초기 데이터 조회 오류:', err.message);
        }
    }

    function handleRealtimeUpdate(payload) {
        const logContainer = document.getElementById('log-container');
        const eventType = payload.eventType;
        const newData = payload.new;
        
        const logItem = document.createElement('div');
        logItem.className = 'log-entry';
        logItem.innerText = `[${eventType}] 유저 ID: ${newData.user_id || 'N/A'} | 현재 깊이: ${newData.depth || 0}m (${new Date().toLocaleTimeString()})`;
        
        logContainer.insertBefore(logItem, logContainer.firstChild);

        if (eventType === 'UPDATE' || eventType === 'INSERT') {
            document.getElementById('user-depth-display').innerText = `현재 깊이: ${newData.depth || 0}m (유저: ${newData.user_id})`;
        }
    }

    function renderInitialList(items) {
        const logContainer = document.getElementById('log-container');
        logContainer.innerHTML = ''; 
        
        items.forEach(item => {
            const logItem = document.createElement('div');
            logItem.className = 'log-entry';
            logItem.innerText = `[기존 기록] 유저 ID: ${item.user_id} | 깊이: ${item.depth}m`;
            logContainer.appendChild(logItem);
        });
        
        if (items[0]) {
            document.getElementById('user-depth-display').innerText = `최근 깊이: ${items[0].depth || 0}m (유저: ${items[0].user_id})`;
        }
    }

    // 모달 및 버튼 이벤트 리스너 정의
    document.querySelectorAll('.modal-backdrop').forEach(m => m.onclick = (e) => { if (e.target === m) m.style.display = 'none'; });
    document.getElementById('introConfirmBtn').onclick = () => document.getElementById('introModalBackdrop').style.display = 'none';
    document.getElementById('cancelCreateBtn').onclick = () => document.getElementById('createModalBackdrop').style.display = 'none';
    document.getElementById('closeViewBtn').onclick = () => document.getElementById('viewModalBackdrop').style.display = 'none';
    document.getElementById('profileCancelBtn').onclick = () => document.getElementById('profileModalBackdrop').style.display = 'none';

    document.getElementById('profileEditBtn').onclick = () => { 
        document.getElementById('dressCodeInput').value = document.getElementById('dressCodeText').textContent; 
        document.getElementById('profileModalBackdrop').style.display = 'flex'; 
    };
    
    document.getElementById('profileSaveBtn').onclick = async () => {
        await supabaseClient.from(PROFILE_TBL).upsert({ id: 1, dress_code: document.getElementById('dressCodeInput').value });
        loadMapData(false);
        document.getElementById('profileModalBackdrop').style.display = 'none';
    };

    // 맵 빈곳 클릭/터치 시 새 핀 등록 프로세스 활성화
    const registerPoint = (clientX, clientY, target) => {
        const rect = target.getBoundingClientRect();
        pendingPoint = { x: (clientX - rect.left) / rect.width, y: (clientY - rect.top) / rect.height };
        document.getElementById('pinName').value = '';
        document.getElementById('pinWeapon').selectedIndex = 0;
        document.getElementById('createModalBackdrop').style.display = 'flex';
    };

    document.getElementById('imageWrap').onmousedown = (e) => { if (e.target.id === 'bgImage') registerPoint(e.clientX, e.clientY, e.target); };
    document.getElementById('imageWrap').ontouchstart = (e) => { if (e.target.id === 'bgImage') registerPoint(e.touches[0].clientX, e.touches[0].clientY, e.target); };

    document.getElementById('saveBtn').onclick = async () => {
        if (!pendingPoint) return;
        await supabaseClient.from(PINS_TBL).insert({ 
            board_id: currentBoard, x_ratio: pendingPoint.x, y_ratio: pendingPoint.y, 
            name: `이름 : ${document.getElementById('pinName').value.trim() || '무명'} / 무기 : ${document.getElementById('pinWeapon').value}` 
        });
        document.getElementById('createModalBackdrop').style.display = 'none';
        loadMapData(false);
    };

    document.getElementById('deleteBtn').onclick = async () => {
        if (!selectedPin || !confirm('이 위치 핀을 삭제하시겠습니까?')) return;
        await supabaseClient.from(PINS_TBL).delete().eq('id', selectedPin.id);
        document.getElementById('viewModalBackdrop').style.display = 'none';
        loadMapData(false);
    };
