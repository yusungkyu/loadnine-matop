    <script src="https://cdn.jsdelivr.net/npm/@supabase/supabase-js@2"></script>
    <style>
        /* 기존 깃허브 원본 레이아웃 및 스타일 유지 */
        body { font-family: 'Malgun Gothic', sans-serif; background-color: #1a1a1a; color: #fff; padding: 20px; }
        .container { max-width: 800px; margin: 0 auto; }
        .panel { border: 1px solid #333; padding: 15px; margin-bottom: 10px; background: #222; border-radius: 6px; }
        .log-entry { font-family: monospace; font-size: 0.9rem; color: #a3be8c; border-bottom: 1px solid #333; padding: 6px 0; }
        .error-msg { color: #bf616a; font-weight: bold; }
    </style>

<body>

<div class="container">
    <h2>🏰 검은마녀의탑 모니터링 대시보드</h2>
    
    <div class="panel">
        <h3>📍 실시간 유저 깊이(Depth) 현황</h3>
        <div id="user-depth-display">데이터를 불러오는 중...</div>
    </div>

    <div class="panel">
        <h3>📜 실시간 트랜잭션 로그</h3>
        <div id="log-container">로그가 이곳에 표시됩니다.</div>
    </div>
</div>

<script>
// 2. DOMContentLoaded 이벤트를 통해 DOM 및 CDN 로드가 완료된 시점에 코드 실행
document.addEventListener('DOMContentLoaded', () => {
    
    // [필수 변수 설정] 본인의 실제 Supabase URL과 Anon KEY로 대체하세요
    const SUPABASE_URL = 'https://your-project-url.supabase.co';
    const SUPABASE_KEY = 'your-anon-key';

    // 스코프 충돌을 방지하기 위해 고유한 변수명 사용
    let supabaseClient;

    try {
        // 라이브러리 로드 상태를 검증한 후 초기화 진행
        if (typeof supabase !== 'undefined') {
            supabaseClient = supabase.createClient(SUPABASE_URL, SUPABASE_KEY);
            console.log('✅ Supabase 클라이언트 초기화 성공');
            
            // 안전하게 앱 초기화 및 실시간 연동 시작
            initApp();
        } else {
            throw new Error('Supabase 라이브러리가 로드되지 않았습니다. CDN 주소를 확인해주세요.');
        }
    } catch (error) {
        console.error('❌ 초기화 실패:', error.message);
        document.getElementById('user-depth-display').innerHTML = `<span class="error-msg">연결 실패: ${error.message}</span>`;
        return;
    }

    // 메인 로직 실행 함수
    function initApp() {
        // 앱이 켜지자마자 기존 데이터 가져오기
        fetchInitialData();
        
        // 3. 실시간 변경 사항 감지 채널 구성 및 SUBSCRIBED 상태 검증
        const trackingChannel = supabaseClient
            .channel('witch-tower-changes')
            .on(
                'postgres_changes',
                {
                    event: '*', // INSERT, UPDATE, DELETE 전체 실시간 감지
                    schema: 'public',
                    table: 'user_tracking' // 본인의 실시간 추적 대상 테이블명으로 수정
                },
                (payload) => {
                    console.log('🔄 실시간 데이터 변동 감지:', payload);
                    handleRealtimeUpdate(payload);
                }
            )
            .subscribe((status) => {
                // 소켓 연결 수립 상태를 안전하게 확인하여 준비되지 않은 접근 방어
                if (status === 'SUBSCRIBED') {
                    console.log('📡 실시간 데이터베이스 채널 구독이 성공적으로 활성화되었습니다.');
                } else if (status === 'TIMED_OUT' || status === 'CHANNEL_ERROR') {
                    console.error(`⚠️ 실시간 채널 연결 지연 또는 실패: ${status}`);
                }
            });
    }

    // 기존 데이터 로드 (초기 새로고침용)
    async function fetchInitialData() {
        try {
            const { data, error } = await supabaseClient
                .from('user_tracking')
                .select('*')
                .order('created_at', { ascending: false })
                .limit(10);

            if (error) throw error;
            
            if (data && data.length > 0) {
                renderInitialList(data);
            } else {
                document.getElementById('user-depth-display').innerText = '현재 기록된 데이터가 없습니다.';
            }
        } catch (err) {
            console.error('❌ 초기 데이터 조회 오류:', err.message);
        }
    }

    // 실시간 이벤트를 받아 화면을 즉시 갱신하는 함수 (새로고침 불필요)
    function handleRealtimeUpdate(payload) {
        const logContainer = document.getElementById('log-container');
        const eventType = payload.eventType;
        const newData = payload.new;
        
        // 4. 실시간 로그 UI 생성 및 추가
        const logItem = document.createElement('div');
        logItem.className = 'log-entry';
        logItem.innerText = `[${eventType}] 유저 ID: ${newData.user_id || 'N/A'} | 현재 깊이: ${newData.depth || 0}m (${new Date().toLocaleTimeString()})`;
        
        // 최신 로그가 맨 위에 쌓이도록 처리
        logContainer.insertBefore(logItem, logContainer.firstChild);

        // 데이터가 업데이트되거나 추가되었을 경우 대시보드 수치 즉시 반영
        if (eventType === 'UPDATE' || eventType === 'INSERT') {
            document.getElementById('user-depth-display').innerText = `현재 깊이: ${newData.depth || 0}m (유저: ${newData.user_id})`;
        }
    }

    // 첫 진입 시 가져온 리스트를 뿌려주는 함수
    function renderInitialList(items) {
        const logContainer = document.getElementById('log-container');
        logContainer.innerHTML = ''; 
        
        items.forEach(item => {
            const logItem = document.createElement('div');
            logItem.className = 'log-entry';
            logItem.innerText = `[기존 기록] 유저 ID: ${item.user_id} | 깊이: ${item.depth}m`;
            logContainer.appendChild(logItem);
        });
        
        // 가장 최신 데이터 대시보드에 표시
        if (items[0]) {
            document.getElementById('user-depth-display').innerText = `최근 깊이: ${items[0].depth || 0}m (유저: ${items[0].user_id})`;
        }
    }
});
</script>
