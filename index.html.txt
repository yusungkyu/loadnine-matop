    <div class="panel">
        <h3>📍 실시간 유저 깊이(Depth) 현황</h3>
        <div id="user-depth-display"></div>
    </div>

    <div class="panel">
        <h3>📜 실시간 트랜잭션 로그 (드래그 정렬 가능)</h3>
        <div id="log-container"></div>

<script>
document.addEventListener('DOMContentLoaded', () => {
    
    // [필수] 본인의 실제 프로젝트 정보로 채워주세요
    const SUPABASE_URL = 'https://your-project-url.supabase.co';
    const SUPABASE_KEY = 'your-anon-key';

    let supabaseClient;

    // 1. Supabase 클라이언트 초기화
    try {
        if (typeof supabase !== 'undefined') {
            supabaseClient = supabase.createClient(SUPABASE_URL, SUPABASE_KEY);
            initApp();
        } else {
            console.error('Supabase 라이브러리 로드 실패');
        }
    } catch (error) {
        console.error('초기화 실패:', error.message);
        return;
    }

    function initApp() {
        fetchInitialData(); 
        setupDragEngine(); 
        
        supabaseClient
            .channel('witch-tower-changes')
            .on(
                'postgres_changes',
                { event: '*', schema: 'public', table: 'user_tracking' },
                (payload) => {
                    handleRealtimeUpdate(payload);
                }
            )
            .subscribe();
    }

    // 2. 초기 데이터 호출
    async function fetchInitialData() {
        try {
            const { data, error } = await supabaseClient
                .from('user_tracking')
                .select('*')
                .order('created_at', { ascending: false })
                .limit(20);

            if (error) throw error;
            renderList(data);
        } catch (err) {
            console.error('데이터 로드 실패:', err.message);
        }
    }

    function renderList(items) {
        const logContainer = document.getElementById('log-container');
        logContainer.innerHTML = ''; 

        if (!items || items.length === 0) return;
        
        items.forEach(item => {
            const logItem = createLogElement(item);
            logContainer.appendChild(logItem);
        });
        
        if (items[0]) updateDepthDisplay(items[0]);
    }

    // 3. 📌 개별 행 엘리먼트 생성 (아바타/탈것 제거 -> 드레스코드 추가)
    function createLogElement(item) {
        const logItem = document.createElement('div');
        logItem.className = 'witch-drag-item'; 
        logItem.id = `pin-${item.id}`; 
        logItem.draggable = true; 

        logItem.addEventListener('dragstart', () => logItem.classList.add('wt-dragging'));
        logItem.addEventListener('dragend', () => logItem.classList.remove('wt-dragging'));
        
        const contentWrap = document.createElement('div');
        contentWrap.className = 'witch-item-content';

        const pinSpan = document.createElement('span');
        pinSpan.className = 'witch-blink-pin';
        pinSpan.innerText = '📌';

        // 📝 [수정 완료] 기존의 아바타, 탈것을 없애고 드레스코드를 출력하도록 명세를 변경했습니다.
        // 만약 DB 컬럼명이 dress_code가 아니라면 item.원하는컬럼명 으로 바꿔주세요.
        const dressCodeText = item.dress_code ? item.dress_code : '미지정';
        
        const textSpan = document.createElement('span');
        textSpan.className = 'witch-text';
        textSpan.innerText = `유저 ID: ${item.user_id} | 깊이: ${item.depth}m | 드레스코드: ${dressCodeText}`;
        
        contentWrap.appendChild(pinSpan);
        contentWrap.appendChild(textSpan);
        
        // 원터치 즉시 삭제 버튼 (확인창 없음)
        const deleteBtn = document.createElement('button');
        deleteBtn.className = 'witch-delete-btn';
        deleteBtn.innerText = '삭제';
        
        deleteBtn.onclick = (e) => {
            e.stopPropagation(); 
            deletePinImmediately(item.id);
        };

        logItem.appendChild(contentWrap);
        logItem.appendChild(deleteBtn);
        return logItem;
    }

    // 4. 드래그 오버 정렬 엔진
    function setupDragEngine() {
        const logContainer = document.getElementById('log-container');
        
        logContainer.addEventListener('dragover', (e) => {
            e.preventDefault(); 
            const draggingItem = document.querySelector('.wt-dragging');
            if (!draggingItem) return;

            const siblings = [...logContainer.querySelectorAll('.witch-drag-item:not(.wt-dragging)')];
            const nextSibling = siblings.find(sibling => {
                const box = sibling.getBoundingClientRect();
                return e.clientY <= box.top + box.height / 2;
            });
            
            if (nextSibling == null) {
                logContainer.appendChild(draggingItem);
            } else {
                logContainer.insertBefore(draggingItem, nextSibling);
            }
        });
    }

    // 5. Supabase 즉시 삭제 기능
    async function deletePinImmediately(id) {
        try {
            const { error } = await supabaseClient
                .from('user_tracking')
                .delete()
                .eq('id', id);

            if (error) throw error;
            console.log(`성공: ID ${id} 삭제 완료.`);
        } catch (err) {
            console.error('삭제 실패:', err.message);
        }
    }

    // 6. 실시간 변동 동기화 핸들러
    function handleRealtimeUpdate(payload) {
        const logContainer = document.getElementById('log-container');
        const eventType = payload.eventType;

        if (eventType === 'INSERT') {
            const newElement = createLogElement(payload.new);
            logContainer.insertBefore(newElement, logContainer.firstChild);
            updateDepthDisplay(payload.new);
        } 
        else if (eventType === 'UPDATE') {
            const target = document.getElementById(`pin-${payload.new.id}`);
            if (target) {
                const dressCodeText = payload.new.dress_code ? payload.new.dress_code : '미지정';
                target.querySelector('.witch-text').innerText = `유저 ID: ${payload.new.user_id} | 깊이: ${payload.new.depth}m | 드레스코드: ${dressCodeText} (수정됨)`;
            }
            updateDepthDisplay(payload.new);
        } 
        else if (eventType === 'DELETE') {
            const target = document.getElementById(`pin-${payload.old.id}`);
            if (target) target.remove(); 
        }
    }

    function updateDepthDisplay(item) {
        document.getElementById('user-depth-display').innerText = `최근 깊이: ${item.depth || 0}m (유저: ${item.user_id})`;
    }
});
